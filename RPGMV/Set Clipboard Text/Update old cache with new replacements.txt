var cache = readFile('translationsCache');
var cache2 = {};

for(var [key,value] of Object.entries(cache)){

  // Replace names with honorifics first
  for (const [keyHon, valueHon] of Object.entries(nameReplacementsWitHonorifics)) {
    let re = new RegExp(keyHon,"g"); // Create regex with variable
    key = key.replace(re, valueHon); // Use regular expression to replace all values and not the first one only
  }

  // Replace words (for names, etc)
  for (const [keyRep, valueRep] of Object.entries(replacements)) {
    let re = new RegExp(keyRep,"g"); // Create regex with variable
    key = key.replace(re, valueRep); // Use regular expression to replace all values and not the first one only
  }

  cache2[key] = value;
}

function getAbsolutePath(){
  let absolutePath = process.cwd();
  if(!absolutePath.includes('www')) absolutePath = absolutePath + '\\www';
  return absolutePath;
}
function getAbsolutePathJson(file){
