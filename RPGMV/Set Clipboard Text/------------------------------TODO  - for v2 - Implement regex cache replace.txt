const { Worker } = require('worker_threads');

function createObjectSearchWorker() {
  const workerCode = `
    const { parentPort, workerData } = require('worker_threads');
    parentPort.on('message', (message) => {
      const { key, object } = message;
      //const result = object.find(obj => obj.hasOwnProperty(key)); //this is deepl not efficient solution
      const result = object[key];
      parentPort.postMessage(result); // will return undefined on not found, maybe change that to null
    });
  `;

  return new Worker(workerCode, { eval: true });
}

function createRegexSearchWorker() {
  const workerCode = `
    const { parentPort, workerData } = require('worker_threads');
    parentPort.on('message', (message) => {
      const { regex, objects } = message;
      const result = objects.find(obj => regex.test(obj));
      // call processCachedTextRegexReplace() before returning (may need to pass JPtext var to worker)
      parentPort.postMessage(result);
    });
  `;

  return new Worker(workerCode, { eval: true });
}

async function searchObjectsAndStrings(key, searchString, objects) {
  const keySearchPromise = new Promise((resolve, reject) => {
    const worker = createObjectSearchWorker();
    worker.on('message', (message) => {
      worker.terminate();
      resolve(message);
    });
    worker.on('error', reject);
    worker.postMessage({ key, objects });
  });

  const regexSearchPromise = new Promise((resolve, reject) => {
    const worker = createRegexSearchWorker();
    worker.on('message', (message) => {
      worker.terminate();
      resolve(message);
    });
    worker.on('error', reject);
    worker.postMessage({ regex: new RegExp(searchString), objects });
  });

  const [keySearchResult, regexSearchResult] = await Promise.all([
    keySearchPromise,
    regexSearchPromise
  ]);

  return keySearchResult || regexSearchResult;
}

async function main() {
  const keyToSearch = 'name';
  const searchString = 'Jo';
  const objects = [
    { name: 'John', age: 30 },
    { name: 'Jane', age: 25 },
    { title: 'Manager', department: 'HR' }
  ];

  try {
    const result = await searchObjectsAndStrings(keyToSearch, searchString, objects);
    console.log(
      `Search for key '${keyToSearch}' or string '${searchString}' returned:`,
      result || 'No match found'
    );
  } catch (error) { // try-catch not needed?
    console.error('An error occurred:', error);
  }
}

main();
