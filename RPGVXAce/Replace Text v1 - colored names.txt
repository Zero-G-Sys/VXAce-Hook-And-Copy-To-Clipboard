# For updating this to v2 (compatibility with message log) see if replacements
# should be before or after escapecharacter function

module ZERO
  # Add a replacement in each line, text to replace can be a regex
  # ex: /word/
  # To match a whole word only put it between \b
  # ex: /\bword\b/
  # use \\1 \\2 etc to get capture group contents (if using "", if using ' ' its \1)
	REPLACEMENTS = {
    #"Text to replace" => "Replacement text",
    #/Text to replace/ => "Replacement text",
    "(Cleff)\n(Åu|\()" => "\eC[14]\\1\n\eC[0]\\2",
  }
end

class Window_Message < Window_Base
  # Overwrite
  def process_all_text
    open_and_wait
    text = convert_escape_characters($game_message.all_text)
    
    ZERO::REPLACEMENTS.each do | base, replacement |
        text.gsub! base, replacement
    end 

    text = convert_escape_characters(text)
    
    pos = {}
    new_page(text, pos)
    process_character(text.slice!(0, 1), text, pos) until text.empty?
  end
end
