 This script will allow to player sets message window visible or unvisible with

 And toogle window ocupacy (make it transparent) while displaying text

 -Press "D" to hide message window and repress "D" to show message

 -Press "C" to turn message window transparent and repress "C" to set default

module EFE
  
  KEY = :Z    # Set hide textbox window and text button
  KEY2 = :C   # Set defualt ok button (show next message)
  KEY3 = :Y   # Set hide textbox window while displaying text button

end

# Global var used for toggling the ocupacy and setting Game_Interpreter
# default window display (needed for dim background)
$globalVar = 0 
$dimBackgroundDisplayed = false

class Window_Message < Window_Base
  @@default_ocupacy = 255
  alias efeberk_window_message_update update
  def update
    efeberk_window_message_update
    @fist_time_entered = true
    
    # Waits for intput and changes ocupacity of window
    # default ocupacity is stored in a constant in case a game doesn't use
    # the default (255). Also sets the global variable for hiding dim backgrounds
    if Input.trigger?(EFE::KEY3)
      if $globalVar == 0
        @@default_ocupacy = self.opacity
        self.opacity = 0
        $globalVar = 1
      elsif $globalVar == 1
        if !$dimBackgroundDisplayed # don't restore ocupacity if it was dim
          self.opacity = @@default_ocupacy
        end
        $globalVar = 0
      end
    end
    
    # Hides textbox completly, makes it visible with same button press
    # or with ok button
    if Input.trigger?(EFE::KEY)
      self.visible = !self.visible
    end
    if Input.press?(EFE::KEY2)
      self.visible = true
    end
    
  end
  
end


# Dim backgrounds don't have a transparency feature, so there is a need to
# modify the core display in the next message via a global variable
class Game_Interpreter

  def command_101
    wait_for_message
    $game_message.face_name = @params[0]
    $game_message.face_index = @params[1]
    $game_message.background = @params[2]
    
    if $game_message.background == 1
      $dimBackgroundDisplayed = true
    else
      $dimBackgroundDisplayed = false
    end
    if $globalVar == 1 # Check if toogled
    	$game_message.background = 2 # Make text window transparent
    end
    
    $game_message.position = @params[3]
    while next_event_code == 401       # Text data
      @index += 1
      $game_message.add(@list[@index].parameters[0])
    end
    case next_event_code
    when 102  # Show Choices
      @index += 1
      setup_choices(@list[@index].parameters)
    when 103  # Input Number
      @index += 1
      setup_num_input(@list[@index].parameters)
    when 104  # Select Item
      @index += 1
      setup_item_choice(@list[@index].parameters)
    end
    wait_for_message
  end
  
end
